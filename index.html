📦 .sandy - Arquivo de construção gerado pelo BloxStudio. Cuidado com aplicativos falsos!

⚙️ .js - Arquivo de mecânicas (player.js). Deve ser criado manualmente com JavaScript.
⚠️ O arquivo .js deve ter extensão .js, não .txt! O nome "player" é opcional, mas a extensão é obrigatória.

🏢 Exemplo de Construção: Sede do SandBlox

Este é um exemplo de arquivo player.js completo que cria a Sede Oficial do SandBlox. Você pode usar como base para criar seus próprios jogos! 

// player.js - SandBlox Headquarters Map (Ultra Stable Edition)

export function initGameMechanics(gameData) {
    console.log("🏢 Carregando a Sede do SandBlox... Direto do escritório dos Devs!");
    if (!gameData || !gameData.scene || !gameData.playerGroup) return;

    const THREE = gameData.THREE;

    // Materiais do Escritório
    const matParede = new THREE.MeshStandardMaterial({ color: 0xe5e7eb, roughness: 0.6 });
    const matMesa = new THREE.MeshStandardMaterial({ color: 0x5c4033, roughness: 0.7 });
    const matPc = new THREE.MeshStandardMaterial({ color: 0x222222, roughness: 0.4 });
    const matTela = new THREE.MeshBasicMaterial({ color: 0x00ff00 });

    // 1. PAREDES DA SEDE
    const paredesSede = [
        [30, 6, 1, 0, 15],   // Parede de Trás
        [30, 6, 1, 0, -15],  // Parede da Frente
        [1, 6, 30, -15, 0],  // Parede Esquerda
        [1, 6, 30, 14, 0]    // Parede Direita
    ];

    paredesSede.forEach(p => {
        const meshParede = new THREE.Mesh(new THREE.BoxGeometry(p[0], p[1], p[2]), matParede);
        meshParede.position.set(p[3], p[1] / 2, p[4]);
        gameData.scene.add(meshParede);
    });

    // 2. DIVISÓRIA DE VIDRO
    const vidroDivisoria = new THREE.Mesh(
        new THREE.BoxGeometry(1, 5, 12),
        new THREE.MeshStandardMaterial({ color: 0xaed6f1, transparent: true, opacity: 0.4 })
    );
    vidroDivisoria.position.set(0, 2.5, 0);
    gameData.scene.add(vidroDivisoria);

    // 3. ESTAÇÕES DE TRABALHO
    const posicoesMesas = [[-8, -6], [-8, 6], [8, -6], [8, 6]];

    posicoesMesas.forEach(pos => {
        const mesa = new THREE.Mesh(new THREE.BoxGeometry(4, 1.2, 2.5), matMesa);
        mesa.position.set(pos[0], 0.6, pos[1]);
        gameData.scene.add(mesa);

        const monitor = new THREE.Mesh(new THREE.BoxGeometry(1.6, 1, 0.2), matPc);
        monitor.position.set(pos[0], 1.7, pos[1] - 0.5);
        gameData.scene.add(monitor);

        const tela = new THREE.Mesh(new THREE.BoxGeometry(1.4, 0.8, 0.05), matTela);
        tela.position.set(pos[0], 1.7, pos[1] - 0.39);
        gameData.scene.add(tela);

        const cpu = new THREE.Mesh(new THREE.BoxGeometry(0.5, 1, 1.2), matPc);
        cpu.position.set(pos[0] + 1.5, 0.5, pos[1]);
        gameData.scene.add(cpu);
    });

    // 4. LOGO DO SANDBLOX
    const logoOuro = new THREE.Mesh(
        new THREE.BoxGeometry(2, 2, 2),
        new THREE.MeshStandardMaterial({ color: 0xf1c40f, metalness: 0.8, roughness: 0.2 })
    );
    logoOuro.position.set(0, 3, -13);
    gameData.scene.add(logoOuro);

    // UI da Sede
    createHQ_UI();

    // Spawn do player
    gameData.playerGroup.position.set(0, 1.2, 0);
}

function createHQ_UI() {
    if (document.getElementById('hq-ui')) return;
    const ui = document.createElement('div');
    ui.id = 'hq-ui';
    ui.style.cssText = `position: absolute; top: 20px; left: 50%; transform: translateX(-50%); background: rgba(30,41,59,0.9); padding: 8px 25px; border-radius: 8px; z-index: 500; border: 1px solid #475569; font-family: sans-serif; color: #fff; text-align: center; box-shadow: 0 4px 15px rgba(0,0,0,0.3);`;
    ui.innerHTML = `
        
🖥️ SandBlox HQ (Sede Oficial)

        
Investigue as salas onde os bugs são criados!

    `;
    document.body.appendChild(ui);
}

export function onGameStart(gameData) {}
export function onGameLoop(gameData) {}
